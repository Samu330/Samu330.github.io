<% // Código para leer el archivo Excel y generar las filas de la tabla // Ejemplo de código en Java utilizando Apache POI try { FileInputStream file = new FileInputStream(new File("ruta_del_archivo_excel.xlsx")); // Obtener el libro de trabajo XSSFWorkbook workbook = new XSSFWorkbook(file); // Obtener la hoja de trabajo XSSFSheet sheet = workbook.getSheetAt(0); // Obtener el número de filas de la hoja de trabajo int rowCount = sheet.getLastRowNum(); // Recorrer las filas y generar las filas de la tabla for (int i = 0; i <= rowCount; i++) { XSSFRow row = sheet.getRow(i); if (row != null) { String nombre = row.getCell(0).getStringCellValue(); int edad = (int) row.getCell(1).getNumericCellValue(); String correo = row.getCell(2).getStringCellValue(); %> <% } } workbook.close(); file.close(); } catch (Exception e) { e.printStackTrace(); } %>

<%= nombre %> <%= edad %> <%= correo %>
